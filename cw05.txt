DECLARE
    numer_max    DEPARTMENTS.DEPARTMENT_ID%TYPE;
    nazwa_dept   DEPARTMENTS.DEPARTMENT_NAME%TYPE := 'EDUCATION';
BEGIN
    SELECT MAX(DEPARTMENT_ID) INTO numer_max FROM DEPARTMENTS;
    
    INSERT INTO DEPARTMENTS (DEPARTMENT_ID, DEPARTMENT_NAME)
    VALUES (numer_max + 10, nazwa_dept);
    
    DBMS_OUTPUT.PUT_LINE('Dodano departament nr: ' || (numer_max + 10));
    COMMIT;
END;
/

    nazwa_dept   DEPARTMENTS.DEPARTMENT_NAME%TYPE := 'EDUCATION';
    nowy_id      DEPARTMENTS.DEPARTMENT_ID%TYPE;
BEGIN
    SELECT MAX(DEPARTMENT_ID) INTO numer_max FROM DEPARTMENTS;
    nowy_id := numer_max + 10;
    
    INSERT INTO DEPARTMENTS (DEPARTMENT_ID, DEPARTMENT_NAME)

    UPDATE DEPARTMENTS
    SET LOCATION_ID = 3000
    WHERE DEPARTMENT_ID = nowy_id;
    
    DBMS_OUTPUT.PUT_LINE('Dodano i zaktualizowano departament nr: ' || nowy_id);
    COMMIT;
END;
/

DECLARE
    CURSOR c_pracownicy IS
        SELECT LAST_NAME, SALARY
        FROM EMPLOYEES
        WHERE DEPARTMENT_ID = 50;
BEGIN
    FOR rec IN c_pracownicy LOOP
        IF rec.SALARY > 3100 THEN
            DBMS_OUTPUT.PUT_LINE(rec.LAST_NAME || ' - nie dawać podwyżki');
        ELSE
            DBMS_OUTPUT.PUT_LINE(rec.LAST_NAME || ' - dać podwyżkę');
        END IF;
    END LOOP;
END;
/

#6
DECLARE
    CURSOR c_pracownicy (
        p_min_sal   EMPLOYEES.SALARY%TYPE,
        p_max_sal   EMPLOYEES.SALARY%TYPE,
        p_imie_part VARCHAR2
    ) IS
        SELECT FIRST_NAME, LAST_NAME, SALARY
        FROM EMPLOYEES
        WHERE SALARY BETWEEN p_min_sal AND p_max_sal
          AND UPPER(FIRST_NAME) LIKE '%' || UPPER(p_imie_part) || '%';
BEGIN
    -- a) widełki 1000-5000, część imienia 'a'
    DBMS_OUTPUT.PUT_LINE('=== 1000-5000, imię zawiera A ===');
    FOR rec IN c_pracownicy(1000, 5000, 'a') LOOP
        DBMS_OUTPUT.PUT_LINE(rec.FIRST_NAME || ' ' || rec.LAST_NAME || ' - ' || rec.SALARY);
    END LOOP;

    -- b) widełki 5000-20000, część imienia 'u'
    DBMS_OUTPUT.PUT_LINE('=== 5000-20000, imię zawiera U ===');
    FOR rec IN c_pracownicy(5000, 20000, 'u') LOOP
        DBMS_OUTPUT.PUT_LINE(rec.FIRST_NAME || ' ' || rec.LAST_NAME || ' - ' || rec.SALARY);
    END LOOP;
END;
/

CREATE OR REPLACE PROCEDURE p_add_job (
    p_job_id    JOBS.JOB_ID%TYPE,
    p_job_title JOBS.JOB_TITLE%TYPE
) AS
BEGIN
    INSERT INTO JOBS (JOB_ID, JOB_TITLE)
    VALUES (p_job_id, p_job_title);
    COMMIT;
    DBMS_OUTPUT.PUT_LINE('Dodano stanowisko: ' || p_job_id);
EXCEPTION
    WHEN DUP_VAL_ON_INDEX THEN
        DBMS_OUTPUT.PUT_LINE('BŁĄD: Stanowisko o tym ID już istnieje.');
    WHEN OTHERS THEN
        DBMS_OUTPUT.PUT_LINE('BŁĄD: ' || SQLERRM);
END;
/

EXEC p_add_job('IT_TEST', 'Tester IT');

CREATE OR REPLACE PROCEDURE p_update_job (
    p_job_id        JOBS.JOB_ID%TYPE,
    p_new_job_title JOBS.JOB_TITLE%TYPE
) AS
    e_no_jobs_updated EXCEPTION;
    PRAGMA EXCEPTION_INIT(e_no_jobs_updated, -20001);
BEGIN
    UPDATE JOBS
    SET JOB_TITLE = p_new_job_title
    WHERE JOB_ID = p_job_id;
    
    IF SQL%ROWCOUNT = 0 THEN
        RAISE_APPLICATION_ERROR(-20001, 'No Jobs updated – nie znaleziono JOB_ID: ' || p_job_id);
    END IF;
    
    COMMIT;
    DBMS_OUTPUT.PUT_LINE('Zaktualizowano stanowisko: ' || p_job_id);
EXCEPTION
    WHEN e_no_jobs_updated THEN
        DBMS_OUTPUT.PUT_LINE('BŁĄD: ' || SQLERRM);
    WHEN OTHERS THEN
        DBMS_OUTPUT.PUT_LINE('BŁĄD: ' || SQLERRM);
END;
/

EXEC p_update_job('IT_TEST', 'Senior Tester IT');
EXEC p_update_job('ZZZ_BRAK', 'Cokolwiek');

CREATE OR REPLACE PROCEDURE p_delete_job (
    p_job_id JOBS.JOB_ID%TYPE
) AS
BEGIN
    DELETE FROM JOBS
    WHERE JOB_ID = p_job_id;
    
    IF SQL%ROWCOUNT = 0 THEN
        RAISE_APPLICATION_ERROR(-20002, 'No Jobs deleted – nie znaleziono JOB_ID: ' || p_job_id);
    END IF;
    
    COMMIT;
    DBMS_OUTPUT.PUT_LINE('Usunięto stanowisko: ' || p_job_id);
EXCEPTION
    WHEN OTHERS THEN
        DBMS_OUTPUT.PUT_LINE('BŁĄD: ' || SQLERRM);
END;
/

EXEC p_delete_job('IT_TEST');
EXEC p_delete_job('ZZZ_BRAK');

CREATE OR REPLACE PROCEDURE p_get_employee_info (
    p_employee_id  IN  EMPLOYEES.EMPLOYEE_ID%TYPE,
    p_last_name    OUT EMPLOYEES.LAST_NAME%TYPE,
    p_salary       OUT EMPLOYEES.SALARY%TYPE
) AS
BEGIN
    SELECT LAST_NAME, SALARY
    INTO p_last_name, p_salary
    FROM EMPLOYEES
    WHERE EMPLOYEE_ID = p_employee_id;
EXCEPTION
    WHEN NO_DATA_FOUND THEN
        DBMS_OUTPUT.PUT_LINE('Nie znaleziono pracownika o ID: ' || p_employee_id);
    WHEN OTHERS THEN
        DBMS_OUTPUT.PUT_LINE('BŁĄD: ' || SQLERRM);
END;
/

DECLARE
    v_nazwisko EMPLOYEES.LAST_NAME%TYPE;
    v_pensja   EMPLOYEES.SALARY%TYPE;
BEGIN
    p_get_employee_info(100, v_nazwisko, v_pensja);
    DBMS_OUTPUT.PUT_LINE('Nazwisko: ' || v_nazwisko || ', Pensja: ' || v_pensja);
END;
/

CREATE OR REPLACE PROCEDURE p_add_employee (
    p_first_name    EMPLOYEES.FIRST_NAME%TYPE,
    p_last_name     EMPLOYEES.LAST_NAME%TYPE,
    p_email         EMPLOYEES.EMAIL%TYPE,
    p_job_id        EMPLOYEES.JOB_ID%TYPE,
    p_salary        EMPLOYEES.SALARY%TYPE,
    p_department_id EMPLOYEES.DEPARTMENT_ID%TYPE DEFAULT NULL,
    p_phone         EMPLOYEES.PHONE_NUMBER%TYPE  DEFAULT NULL,
    p_hire_date     EMPLOYEES.HIRE_DATE%TYPE      DEFAULT SYSDATE,
    p_manager_id    EMPLOYEES.MANAGER_ID%TYPE     DEFAULT NULL
) AS
BEGIN
    IF p_salary > 20000 THEN
        RAISE_APPLICATION_ERROR(-20003, 'Wynagrodzenie przekracza dopuszczalny limit 20000.');
    END IF;

    INSERT INTO EMPLOYEES (
        EMPLOYEE_ID, FIRST_NAME, LAST_NAME, EMAIL,
        PHONE_NUMBER, HIRE_DATE, JOB_ID, SALARY,
        DEPARTMENT_ID, MANAGER_ID
    ) VALUES (
        (SELECT NVL(MAX(EMPLOYEE_ID), 200) + 1 FROM EMPLOYEES),
        p_first_name, p_last_name, p_email,
        p_phone, p_hire_date, p_job_id, p_salary,
        p_department_id, p_manager_id
    );

    COMMIT;
    DBMS_OUTPUT.PUT_LINE('Dodano pracownika: ' || p_first_name || ' ' || p_last_name);
EXCEPTION
    WHEN OTHERS THEN
        DBMS_OUTPUT.PUT_LINE('BŁĄD: ' || SQLERRM);
END;/

EXEC p_add_employee('Jan', 'Kowalski', 'JKOWALSKI', 'IT_PROG', 8000, 60);
EXEC p_add_employee('Anna', 'Nowak', 'ANOWAK', 'IT_PROG', 25000, 60);
