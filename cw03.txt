SELECT employee_id, first_name, last_name, salary,
RANK() OVER(ORDER BY salary DESC) as rankingpensji
FROM EMPLOYEES;

SELECT employee_id,first_name,last_name,salary,
SUM(salary) OVER () as sumapensji
FROM EMPLOYEES;

CREATE TABLE sales as SELECT * FROM hr.sales;
CREATE TABLE products as SELECT * FROM hr.products;


SELECT e.last_name, p.product_name, SUM(s.price*s.quantity) OVER (PARTITION BY s.employee_id) as skumulowane,
RANK() OVER (ORDER BY s.price * s.quantity DESC) as ranking
FROM sales s
JOIN employees e
    ON s.employee_id = e.employee_id
JOIN products p
    ON s.product_id = p.product_id;


SELECT e.last_name, p.product_name, s.price, 
COUNT(*) OVER ( PARTITION BY s.product_id, s.sale_date) AS ilosc,
SUM(s.price *s.quantity) OVER (PARTITION BY s.product_id, s.sale_date) AS dziennasuma,
LAG(s.price) OVER (PARTITION BY s.product_id ORDER BY s.sale_date) AS wczesniejsza,
LEAD(s.price)  OVER (PARTITION BY s.product_id ORDER BY s.sale_date) AS nastepna
FROM sales s
JOIN employees e
    ON s.employee_id=e.employee_id
JOIN products p 
    ON s.product_id =p.product_id;


SELECT p.product_name,s.price, SUM(s.price * s.quantity) OVER (PARTITION BY TRUNC(s.sale_date, 'MM')) AS miesiecznie,
SUM(s.price*s.quantity) OVER(PARTITION BY p.product_id, TRUNC (s.sale_date,'MM') ORDER BY s.sale_date) AS miesieczniasuma
FROM sales s
JOIN products p
    ON s.product_id=p.product_id;


SELECT p.product_name,p.product_category,s22.price as cena_2022,
s23.price as cena_2023, s23.price -s22.price as roznica_cen
FROM sales s22
JOIN sales s23
    ON s22.product_id = s23.product_id
    AND TO_CHAR(s22.sale_date, 'MM-DD') = TO_CHAR(s23.sale_date,'MM-DD')
    AND EXTRACT(YEAR FROM s22.sale_date)=2022
    AND EXTRACT(YEAR FROM s23.sale_date)=2023
JOIN products p
    ON s22.product_id =p.product_id;


SELECT p.product_category, p.product_name,s.price,
MIN(s.price) OVER (PARTITION BY p.product_category) AS minimalna,
MAX(s.price) OVER (PARTITION BY p.product_category) AS maksymalna,
MAX(s.price) OVER (PARTITION BY p.product_category) - MIN(s.price) OVER (PARTITION BY p.product_category) AS rozncia
FROM sales s
JOIN products p
    ON s.product_id = p.product_id;


SELECT p.product_name,s.sale_date,s.price,
AVG(s.price) OVER (PARTITION BY s.product_id ORDER BY s.sale_date ROWS BETWEEN 1 PRECEDING AND 1 FOLLOWING) AS sredniakroczaca
FROM sales s
JOIN products p
    ON s.product_id = p.product_id;


SELECT p.product_name,p.product_category,
RANK() OVER (PARTITION BY p.product_category ORDER BY s.price DESC) as rankingcen,
ROW_NUMBER() OVER (PARTITION BY p.product_category ORDER BY s.price DESC) as numerwiersza,
DENSE_RANK() OVER (PARTITION BY p.product_category ORDER BY s.price DESC) as denseranking
FROM sales s
JOIN products p
    ON s.product_id = p.product_id;


SELECT e.last_name, p.product_name, 
SUM(s.price * s.quantity) OVER (PARTITION BY s.employee_id ORDER BY s.sale_date) AS wartoscsprzedazy,
RANK() OVER (ORDER BY s.price * s.quantity DESC) as rankingwartosci
FROM sales s
JOIN employees e
    ON s.employee_id = e.employee_id
JOIN products p 
    ON s.product_id = p.product_id;


SELECT e.first_name,e.last_name,j.job_title
FROM employees e
JOIN jobs j
    ON e.job_id = j.job_id
WHERE e.employee_id in (SELECT employee_id FROM sales);
